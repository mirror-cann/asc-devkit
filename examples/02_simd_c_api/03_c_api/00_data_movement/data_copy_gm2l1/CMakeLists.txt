cmake_minimum_required(VERSION 3.16)

set(CMAKE_ASC_RUN_MODE "npu" CACHE STRING "Run mode: npu, sim")
set(CMAKE_ASC_ARCHITECTURES "dav-3510" CACHE STRING "NPU architecture: dav-3510")

find_package(ASC REQUIRED)

project(c_api_data_copy_gm2l1 LANGUAGES ASC CXX)

if(NOT CMAKE_ASC_ARCHITECTURES STREQUAL "dav-3510")
    message(FATAL_ERROR "This C API sample only supports CMAKE_ASC_ARCHITECTURES=dav-3510.")
endif()

if(NOT CMAKE_ASC_RUN_MODE MATCHES "^(npu|sim)$")
    message(FATAL_ERROR "CMAKE_ASC_RUN_MODE must be npu or sim, but got ${CMAKE_ASC_RUN_MODE}.")
endif()

# 场景参数（通过命令行传入）
# SCENARIO_NUM: 场景编号
#   1 - 输入格式Nz，输入数据类型half
#   2 - 输入格式ND，输入数据类型half
#   3 - 输入格式DN，输入数据类型half
#   4 - 输入格式ND，输入数据类型half，使能Vector量化
set(SCENARIO_NUM "1" CACHE STRING "Scene number (1-4)")
if(NOT SCENARIO_NUM MATCHES "^[1-4]$")
    message(FATAL_ERROR "SCENARIO_NUM must be 1, 2, 3, or 4, but got ${SCENARIO_NUM}")
endif()

add_executable(demo data_copy_gm2l1.asc)

add_definitions(-DSCENARIO_NUM=${SCENARIO_NUM})
target_compile_options(demo PRIVATE
    $<$<COMPILE_LANGUAGE:ASC>:--npu-arch=${CMAKE_ASC_ARCHITECTURES}>
)
